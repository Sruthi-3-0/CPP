pipeline {
    agent any

    stages {

        stage('Checkout') {
            steps {
                checkout scm
            }
        }

        stage('Terraform Init') {
            steps {
                dir('.') {
                    bat 'terraform init'
                }
            }
        }

        stage('Terraform Validate') {
            steps {
                bat 'terraform validate'
            }
        }

        stage('tfsec Scan') {
            steps {
                bat '''
                docker run --rm ^
                -v %cd%:/src ^
                aquasec/tfsec /src || exit 0
                '''
            }
        }

        stage('Checkov Scan') {
            steps {
                bat '''
                docker run --rm ^
                -v %cd%:/tf ^
                bridgecrew/checkov -d /tf || exit 0
                '''
            }
        }

        stage('Docker Build') {
            steps {
                bat 'docker build -f DOCKERFILE.txt -t cpp-app .'
            }
        }

        stage('Terraform Apply') {
            steps {
                bat 'terraform apply -auto-approve'
            }
        }
    }
}
